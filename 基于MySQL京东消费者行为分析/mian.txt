#用户活跃特征分析（每周PV/UV）
#按周统计一下浏览量和访客数，看看每周的人均浏览量怎么样
#PV/UV比值越高说明用户逛得越深，比值低的话可能用户进来就走了

"""
WITH t1 as (
SELECT
    action_date,
    FLOOR(DATEDIFF(action_date, '2018-02-01') / 7) AS num,
    COUNT(customer_id) AS pv,
    COUNT(DISTINCT customer_id) AS uv
FROM
    jd_database_new
GROUP BY action_date
)

SELECT
    CONCAT(start_date, ' 至 ', end_date) AS date,
    PV AS "每周PV",
    UV AS "每周UV",
    ROUND(PV / UV, 2) AS "pv/uv"
FROM (SELECT
        num,
DATE_ADD((SELECT MIN(action_date) FROM jd_database_new), INTERVAL (num 7) DAY AS start_date,
DATE_ADD((SELECT MIN(action_date) FROM jd_database_new), INTERVAL (num * 7 + 6) DAY >
SELECT MAX(action_date) FROM jd_database_new,
SELECT MAX(action_date) FROM jd_database_new,
DATE_ADD((SELECT MIN(action_date) FROM jd_database_new), INTERVAL (num * 7 + 6) DAY) AS end_date,
        SUM(pv) PV,
        SUM(uv) UV
     FROM t1
     GROUP BY num
     )t2;

"""

#RFM用户分群
#根据用户的最近购买时间、购买次数、购买多样性，把用户分成8类
#重要价值用户是又近又频繁又多样的要重点维护，一般挽留用户流失风险大要想办法召回

"""
WITH w1 AS (
    SELECT
        customer_id,
        DATEDIFF((SELECT MAX(action_date) FROM jd_database_new), MAX(action_date)) AS recency,
        COUNT(*) AS frequency,
        COUNT(DISTINCT category) AS monetary_m1,
        COUNT(DISTINCT type) AS monetary_m2,
        COUNT(DISTINCT shop_id) AS monetary_m3
    FROM jd_database_new
    WHERE type = 'Order'
    GROUP BY customer_id
)
SELECT
    CASE
        WHEN r_score >= 4 AND f_score >= 4 AND m_score >= 4 THEN '重要价值用户'
        WHEN r_score < 4  AND f_score >= 4 AND m_score >= 4 THEN '重要保持用户'
        WHEN r_score >= 4 AND f_score < 4  AND m_score >= 4 THEN '重要发展用户'
        WHEN r_score < 4  AND f_score < 4  AND m_score >= 4 THEN '重要挽留用户'
        WHEN r_score >= 4 AND f_score >= 4 AND m_score < 4  THEN '一般价值用户'
        WHEN r_score < 4  AND f_score >= 4 AND m_score < 4  THEN '一般保持用户'
        WHEN r_score >= 4 AND f_score < 4  AND m_score < 4  THEN '一般发展用户'
        ELSE '一般挽留用户'
    END AS '用户类型',
    COUNT(customer_id) AS cnt
FROM (
    SELECT
        customer_id,
        NTILE(6)OVER(ORDER BY recency DESC) AS r_score,
        NTILE(6)OVER(ORDER BY frequency) AS f_score,
        (NTILE(6)OVER(ORDER BY monetary_m1) + NTILE(6)OVER(ORDER BY monetary_m2) + NTILE(6)OVER(ORDER BY monetary_m3)) / 3 AS m_score
    FROM w1
    )w2
GROUP BY 用户类型
ORDER BY 用户类型;
""" )

#用户购买转化漏斗分析
#看看用户从浏览到最后付款，整个过程每一步流失多少
#结果会显示 浏览->加购->下单->支付 各个环节的人数和转化率
#可以知道哪个环节问题最大，比如加购很多但下单少，就要去优化购物车页面

"""
SELECT
    type,
    COUNT(*) AS cnt,
    CONCAT(ROUND(COUNT(*) / LAG(COUNT(*)) OVER (ORDER BY FIELD(type, 'Browse', 'Cart', 'Order', 'Pay')) * 100, 2), '%') AS conversion_rate
FROM
    jd_database_new
WHERE
    type IN ('Browse', 'Cart', 'Order', 'Pay')
GROUP BY type
ORDER BY FIELD(type, 'Browse', 'Cart', 'Order', 'Pay');
"""

#用户购买周期分析
#看看用户一般多久来买一次东西，方便知道什么时候该提醒用户回来买东西
#可以按会员等级看看差异，高级会员是不是购买更频繁

"""
WITH purchase间隔 AS (
    SELECT
        customer_id,
        action_date,
        LAG(action_date) OVER (PARTITION BY customer_id ORDER BY action_date) AS last_purchase_date
    FROM jd_database_new
    WHERE type = 'Order'
)
SELECT
    AVG(DATEDIFF(action_date, last_purchase_date)) AS avg_purchase_interval,
    customer_level,
    COUNT(*) AS cnt
FROM purchase间隔 t1
JOIN jd_database_new t2 ON t1.customer_id = t2.customer_id
WHERE last_purchase_date IS NOT NULL
GROUP BY customer_level
ORDER BY avg_purchase_interval;
"""

#用户价值分层分析
#按用户的购买次数把用户分成5个等级，看看各等级的分布情况
#高等级用户虽然人少，但肯定是下单最多的，要重点维护

"""
WITH user_value AS (
    SELECT
        customer_id,
        COUNT(*) AS order_cnt,
        COUNT(DISTINCT category) AS category_cnt,
        NTILE(5) OVER (ORDER BY COUNT(*)) AS value_level
    FROM jd_database_new
    WHERE type = 'Order'
    GROUP BY customer_id
)
SELECT
    CASE value_level
        WHEN 5 THEN '高价值用户'
        WHEN 4 THEN '次高价值用户'
        WHEN 3 THEN '中等价值用户'
        WHEN 2 THEN '次低价值用户'
        ELSE '低价值用户'
    END AS user_level,
    COUNT(customer_id) AS user_cnt,
    SUM(order_cnt) AS total_orders,
    ROUND(AVG(order_cnt), 2) AS avg_orders,
    CONCAT(ROUND(COUNT(customer_id) / (SELECT COUNT(DISTINCT customer_id) FROM user_value) * 100, 2), '%') AS proportion
FROM user_value
GROUP BY value_level
ORDER BY value_level DESC;
"""

#复购率分析
#看看用户买了之后有多大概率会再回来买，30天内复购率大概是什么水平
#复购率高说明用户认可平台，愿意再来；复购率低就要想办法召回用户了

"""
WITH first_purchase AS (
    SELECT
        customer_id,
        MIN(action_date) AS first_date
    FROM jd_database_new
    WHERE type = 'Order'
    GROUP BY customer_id
),
repeat_purchase AS (
    SELECT
        t1.customer_id,
        t1.first_date,
        COUNT(DISTINCT t2.action_date) AS repeat_cnt
    FROM first_purchase t1
    JOIN jd_database_new t2 ON t1.customer_id = t2.customer_id
        AND t2.action_date > t1.first_date
        AND t2.action_date <= DATE_ADD(t1.first_date, INTERVAL 30 DAY)
    WHERE t2.type = 'Order'
    GROUP BY t1.customer_id, t1.first_date
)
SELECT
    COUNT(DISTINCT fp.customer_id) AS total_buyers,
    COUNT(DISTINCT rp.customer_id) AS repeat_buyers,
    CONCAT(ROUND(COUNT(DISTINCT rp.customer_id) / COUNT(DISTINCT fp.customer_id) * 100, 2), '%') AS repeat_rate_30d
FROM first_purchase fp
LEFT JOIN repeat_purchase rp ON fp.customer_id = rp.customer_id;
"""

#用户行为路径分析
#看看用户一般是怎么逛的，从哪个行为跳到哪个行为最多
#可以发现用户常见的逛法，然后优化那些容易流失的路径

"""
WITH user_action_sequence AS (
    SELECT
        customer_id,
        action_date,
        type,
        LAG(type) OVER (PARTITION BY customer_id ORDER BY action_date, action_id) AS prev_action,
        LEAD(type) OVER (PARTITION BY customer_id ORDER BY action_date, action_id) AS next_action
    FROM jd_database_new
)
SELECT
    prev_action,
    type AS current_action,
    next_action,
    COUNT(*) AS cnt
FROM user_action_sequence
WHERE prev_action IS NOT NULL AND next_action IS NOT NULL
GROUP BY prev_action, type, next_action
ORDER BY cnt DESC
LIMIT 30;
"""
